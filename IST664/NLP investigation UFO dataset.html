# Convert text data into numerical vectors using TF-IDF
tfidf_vectorizer = TfidfVectorizer()
tfidf_matrix = tfidf_vectorizer.fit_transform(sample)

# Apply k-means clustering
k = 3 # Number of clusters
kmeans = KMeans(n_clusters=k)
kmeans.fit(tfidf_matrix)

# Reduce the dimensionality of TF-IDF matrix using PCA
pca = PCA(n_components=2)
tfidf_matrix_reduced = pca.fit_transform(tfidf_matrix.toarray())

# Plot clusters
plt.figure(figsize=(8, 6))
for i in range(k):
    plt.scatter(tfidf_matrix_reduced[kmeans.labels_ == i, 0],
                tfidf_matrix_reduced[kmeans.labels_ == i, 1],
                label=f'Cluster {i+1}')
plt.title('K-Means Clustering')
plt.xlabel('Principal Component 1')
plt.ylabel('Principal Component 2')
plt.legend()
plt.grid(True)
plt.show()
